julia> # this function calculates DO downstream from some initial conditions and other parameters
       function do_calc(x, C0, B0, N0, ka, kn, kc, Cs, U)
           B = B0 .* exp.(-kc .* x ./ U)
           N = N0 .* exp.(-kn .* x ./ U)
           alpha1 = exp.(-ka .* x ./ U)
           alpha2 = (kc/(ka-kc)) .* (exp.(-kc .* x / U) - exp.(-ka .* x / U))
           alpha3 = (kn/(ka-kn)) .* (exp.(-kn .* x / U) - exp.(-ka .* x / U))
           C = Cs .* (1 .- alpha1) + (C0 .* alpha1) - (B0 .* alpha2) - (N0 .* alpha3)
           return (C, B, N)
       end

julia> # this function takes in some other parameters (treatment efficients E1 and E2, length l), and  runs the full simulation
       function do_simulate(E1, E2, l; C0=7.5, B0 = 5, N0 = 5, ka = 0.55, kc = 0.35, kn = 0.25, Cs = 10, U = 6)
       
         C = zeros(l);
         N = zeros(l);
         B = zeros(l);
       
         C[1] = mixed_concentration(C0, 5, 100000, 10000);
         B[1] = mixed_concentration(B0, (1-E1)*50, 100000, 10000);
         N[1] = mixed_concentration(N0, (1-E1)*35, 100000, 10000);
       
         # at some point a loop is necessary to update the conditions at every step. The size of the step actually doesn't matter for this problem, so long as the change in initial conditions at the second stream is taken into account.
         for t = 2:l
           if t < 16
             C[t], B[t], N[t] = do_calc(t-1, C[1], B[1], N[1], ka, kn, kc, Cs, U);
           elseif t == 16
           Cin, Bin, Nin = do_calc(t-1, C[1], B[1], N[1], ka, kn, kc, Cs, U);
             C[t] = mixed_concentration(Cin, 5, 110000, 15000);
             B[t] = mixed_concentration(Bin, (1-E2)*45, 110000, 15000);
             N[t] = mixed_concentration(Nin, (1-E2)*35, 110000, 15000);
           else
             C[t], B[t], N[t] = do_calc(t-16, C[16], B[16], N[16], ka, kn, kc, Cs, U);
           end
         end
       
         return C, B, N
       end

Problem 1.8

In this case, the correlation actually decreases the probability that the solution will keep the system in compliance, as generally if one type of DO is higher, the other will be as well, whereas in the uncorrelated case either CBOD or NBOD could be high enough to cause the DO levels to drop.

In both cases, though, this strategy fails to keep us in compliance more often than not, so we might want to seek out a treatment plan that has a higher margin of safety instead of doing as little as possible in the expected case.
